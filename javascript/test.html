1. 1부터 100까지 출력하기(for문, while, do while)

{
    for (let i = 1; i <= 100; i++) {
        console.log(i);
    }
    // 1,2,3,4...100
}
{
    let num = 1;

    while (num <= 100) {
        console.log(num);
        num++;
    }
}
{
    let num = 1;

    do {
        console.log(num);
        num++;
    } while (num <= 100);
}

02. 1부터 100까지 짝수만 출력하기(for문, if문)

{
    for (let i = 1; i <= 100; i++) {
        if (i % 2 == 0) {
            console.log(i);
        }
    }
}

03. 1부터 100까지 짝수는 파란색, 홀수는 빨간색 출력하기

{
    {
        for (let i = 1; i <= 100; i++) {
            if (i % 2 == 0) {
                document.write("" + i + "");
            } else {
                document.write("" + i + "");
            }
        }
    }
}

04. 중첩 for문을 이용해서 구구단 출력하기

{
    for (let i = 2; i <= 9; i++) {
        for (let j = 2; j <= 9; j++) {
            console.log(i + "*" + j + "=" + i * j);
        }
    }
}

05. 중첩 for문을 이용해서 8단 출력하기

{
    for (let i = 8; i <= 8; i++) {
        for (let j = 1; j <= 9; j++) {
            console.log(i + "*" + j + "=" + i * j);
        }
    }
}
